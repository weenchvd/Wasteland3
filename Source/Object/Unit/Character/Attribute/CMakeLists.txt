cmake_minimum_required(VERSION 3.20)

# specify the C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED TRUE)

#**********************************************************************
# create Flatbuffers files for AttributeReference
set(ATTRIBUTE_REF_FB_FILE_STEM "attributeReferenceFB")
set(ATTRIBUTE_REF_FB_BIN_FILE_IDENTIFIER "ATTR")
set(ATTRIBUTE_REF_FB_BIN_FILE_EXTENSION "${FB_BIN_FILE_EXTENSION}")
set(ATTRIBUTE_REF_FB_BIN_FILE_NAME "${ATTRIBUTE_REF_FB_FILE_STEM}.${ATTRIBUTE_REF_FB_BIN_FILE_EXTENSION}")
file(REAL_PATH "${FB_DATA_DIR_NAME}/${ATTRIBUTE_REF_FB_BIN_FILE_NAME}" ATTRIBUTE_REF_FB_BIN_FILE)
file(${DEFAULT_CMAKE_FILE_HASH_FUNC} "${ATTRIBUTE_REF_FB_BIN_FILE}" ATTRIBUTE_REF_FB_BIN_FILE__HASH)
file(RELATIVE_PATH ATTRIBUTE_REF_FB_BIN_FILE__REL_PATH_FROM_CSD
                   "${CMAKE_SOURCE_DIR}"
                   "${ATTRIBUTE_REF_FB_BIN_FILE}"
)
cmake_path(HASH ATTRIBUTE_REF_FB_BIN_FILE__REL_PATH_FROM_CSD
                ATTRIBUTE_REF_FB_BIN_FILE__REL_PATH_FROM_CSD__HASH
)
string(CONCAT ATTRIBUTE_REF_FB_BIN_FILE_NAME__OUT
  "${ATTRIBUTE_REF_FB_FILE_STEM}_${ATTRIBUTE_REF_FB_BIN_FILE__REL_PATH_FROM_CSD__HASH}."
  "${ATTRIBUTE_REF_FB_BIN_FILE_EXTENSION}"
)
file(TO_NATIVE_PATH
  "${ASSETS__DIR__REL_PATH_FROM_W3EXEC}/${ATTRIBUTE_REF_FB_BIN_FILE_NAME__OUT}"
  ATTRIBUTE_REF_FB_BIN_FILE__NATIVE_REL_PATH
)

cmake_language(CALL create_flatbuffers_data_files
  "${ATTRIBUTE_REF_FB_BIN_FILE_IDENTIFIER}"
  "${ATTRIBUTE_REF_FB_BIN_FILE_EXTENSION}"
  "${ATTRIBUTE_REF_FB_FILE_STEM}"
)
cmake_language(CALL distribute_file
  "${ATTRIBUTE_REF_FB_BIN_FILE}"
  "${ASSETS__DIR__REL_PATH_FROM_W3EXEC}"
  "${ATTRIBUTE_REF_FB_BIN_FILE_NAME__OUT}"
)
